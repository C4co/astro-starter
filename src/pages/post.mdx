```javascript
function pow(x, n) {
  let result = 1

  // multiply result by x n times in the loop
  for (let i = 0; i < n; i++) {
    result *= x
  }

  return result
}

alert(pow(2, 3)) // 8
```

```javascript
function marry(man, woman) {
  woman.husband = man
  man.wife = woman

  return {
    father: man,
    mother: woman,
  }
}

let family = marry(
  {
    name: 'John',
  },
  {
    name: 'Ann',
  }
```
